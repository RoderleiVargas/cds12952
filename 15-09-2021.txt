#include <stdio.h>

int main()
{
int i, nro, soma=0, produto =  1;

printf("\n Digite um número natural: ");
scanf("%d", &nro);

    for (i = 1; i <= nro; i++)
    {
        soma += i;
        produto *= i;
    }
    printf("\nSoma = %d", soma);
    printf("\nProduto = %d", produto);
    
    return 0;
}

# include <stdio.h>

int main()
{
    int n, i, j;

    printf("Digite o número de ramos: ");
    scanf("%d", &n);

    for(i = 0; i <= n; i++)
    {
        for(j = 0; j <= i; j++)
        {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
