Tipo de Variáveis

%d = inteiro = int
%f = numero com virgula - float
%c = unico caracter = char
%s = string caracter = char colocar o tamanho entre colchetes
& = utilizado somente quando são nros ou um unico caracter

----------------- EXEMPLO DE INTEIRO

#include <stdio.h>

int codigo;

int main()
{
    printf("\nDigite o codigo: ");
    scanf("%d", &codigo);

    return 0;
}
